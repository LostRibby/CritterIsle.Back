pour les recherches d'animaux voici les exemples de liens : 

🦋insectes = http://localhost:3000/Bugs
ajouter filtre :

location = sol, rocher, souche d'arbre, arbres, palmier, fleurs.

season = hiver, printemps, été, automne.

exemple de lien avec les 2 filtres :

http://localhost:3000/Bugs?location=sol&season=hiver

🦈poisons = http://localhost:3000/Fishes

ajouter filtre:

location = mer, ponton, étang, rivière, cascade

season = hiver, printemps, été, automne
       
  exemple de lien avec les 2 filtres : 
  
 http://localhost:3000/Fishes?location=rivière&season=hiver

🪼créatures marine = http://localhost:3000/Creatures 

  ajouter filtre : 
  
  speed= immobile, très_lent, lent, moyen, rapide, très_rapide
       
  season = hiver, printemps, été, automne

  exemple de lien avec les 2 filtres :
  
  http://localhost:3000/Creatures?speed=très_lent&season=hiver
